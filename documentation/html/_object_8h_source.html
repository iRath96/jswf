64  TraitMatch getTraitByName(const Multiname &name) {
65  Class *k = klass;
66  while(k != NULL) {
67  // Search instance traits.
68  for(auto it = k->iinfo.traits.begin(); it != k->iinfo.traits.end(); ++it)
69  if(*(*it)->name == name)
70  return { .isStatic = false, .trait = it->get(), .dataStore = &traitMap };
71 
72  // Search class traits.
73  for(auto it = k->cinfo.traits.begin(); it != k->cinfo.traits.end(); ++it)
74  if(*(*it)->name == name)
75  return { .isStatic = true, .trait = it->get(), .dataStore = &klass->traitMap };
76 
77  k = k->parent;
78  } return { .trait = NULL, .dataStore = NULL };
79  }
