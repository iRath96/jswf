157  void read(io::GenericReader *reader, ABCFile *file, IConstantSource<T> *source, std::stack<std::shared_ptr<T>> &stack) {
158 #define __pop (top = stack.top(), stack.pop(), top)
159 
160  pos = reader->pos;
161 
162  code = (Opcode::Code)reader->readU8();
163  opcode = opcodes[code];
164 
165  std::shared_ptr<T> top; // for __pop
166 
167  // opcode argument order: (each part is optional)
168  // multiname_index, arg_count, register_index, branch_offset, pool_index, immediate
169  // opcode stack order: (each part is optional)
170  // value1, multiname_ns, multiname_name, value2, arg0...argN
171 
172  if(opcode.flags & Opcode::HasMultinameFlag) multinameIndex = reader->readVU30();
173 
174  if(opcode.flags & Opcode::HasArgumentCountFlag) {
175  argumentCount = reader->readVU30();
176  for(uint32_t i = 0; i < argumentCount; ++i) arguments.insert(arguments.begin(), __pop);
177  }
178 
179  if(opcode.stackPop > 1) value2 = __pop;
180 
181  if(opcode.flags & Opcode::HasMultinameFlag) {
182  multiname = *file->constantPool.multinames[multinameIndex];
183  if(opcode.flags & Opcode::MultinameIsQNameFlag) {
184  if(!multiname.isQualified())
185  throw "Error error error.";
186  } else {
187  if(!multiname.hasName) multiname.setName(__pop->coerce_s());
188  if(!multiname.hasNS) multiname.setNS(std::make_shared<Namespace>(__pop->coerce_ns()));
189  }
190  }
191 
192  if(opcode.flags & Opcode::HasRegisterFlag) registerIndex = reader->readVU30();
193  if(opcode.flags & Opcode::IsBranchFlag) branchOffset = reader->readS24();
194 
195  if((opcode.flags & Opcode::HasPoolConstantFlag) == Opcode::HasPoolConstantFlag)
196  constant = source->getPoolConstant(opcode.constantKind, file, poolIndex = reader->readVU30());
197  else if(opcode.flags & Opcode::HasConstantFlag)
198  constant = source->getConstant(opcode.constantKind);
199 
200  if(opcode.stackPop > 0) value1 = __pop;
201 
202  if(opcode.flags & Opcode::HasImmediateValueFlag) {
203  switch(opcode.constantKind) {
204  case ConstantKind::ByteKind : immediateInt = reader->readS8(); break;
205  case ConstantKind::ShortKind: immediateInt = reader->readVU30(); break;
206  default: immediateInt = 0;
207  }
208 
209  constant = source->getImmediateConstant(opcode.constantKind, immediateInt);
210  }
