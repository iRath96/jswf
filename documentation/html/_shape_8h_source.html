80  void polygonize() {
81  std::vector<Segment> flatSegments;
82 
83 #define SUBBEZIER 16
84  for(auto seg = segments.begin(); seg != segments.end(); ++seg) {
85  if(seg->isCurved) {
86  sb_t x = seg->x0, y = seg->y0;
87  for(int i = 1; i <= SUBBEZIER; ++i) {
88  double t = i / (double)SUBBEZIER;
89 
90 #define MORPH(a, b, t) (a + (b - a) * t)
91  sb_t xa = MORPH(seg->x0, seg->cx, t), ya = MORPH(seg->y0, seg->cy, t);
92  sb_t xb = MORPH(seg->cx, seg->x1, t), yb = MORPH(seg->cy, seg->y1, t);
93  sb_t xx = MORPH(xa, xb, t), yy = MORPH(ya, yb, t);
94 #undef MORPH
95 
96  Segment s = *seg;
97  s.isCurved = false;
98  s.x0 = x ; s.y0 = y ;
99  s.x1 = xx; s.y1 = yy;
100 
101  flatSegments.push_back(s);
102 
103  x = xx; y = yy;
104  }
105  } else
106  flatSegments.push_back(*seg);
107  }
108 
109  // Magic
110 
111  std::map<styles::FillStylePtr, std::vector<Edge>> rawEdges;
112  for(auto it = flatSegments.begin(); it != flatSegments.end(); ++it) {
113  if(it->fillStyle0 != NULL) {
114  Edge e = { .a = { .x = it->x0, .y = it->y0 }, .b = { .x = it->x1, .y = it->y1 } };
115  if(e.a != e.b) rawEdges[it->fillStyle0].push_back(e);
116  }
117 
118  if(it->fillStyle1 != NULL) { // but reverse.
119  Edge e = { .b = { .x = it->x0, .y = it->y0 }, .a = { .x = it->x1, .y = it->y1 } };
120  if(e.a != e.b) rawEdges[it->fillStyle1].push_back(e);
121  }
122  }
123 
124  // Weeeee
125 
126  polygons.clear();
127  for(auto it = rawEdges.begin(); it != rawEdges.end(); ++it) {
128  std::vector<Edge> &pEdges = it->second;
129 
130  Point firstPoint = pEdges[0].a;
131  Point searchPoint = pEdges[0].a;
132 
133  Polygon p;
134 
135  while(!pEdges.empty()) {
136  bool foundNextEdge = false;
137  Edge nextEdge;
138 
139  for(auto edge = pEdges.begin(); edge != pEdges.end(); ++edge)
140  if(edge->a == searchPoint) {
141  foundNextEdge = true;
142  nextEdge = *edge;
143  pEdges.erase(edge);
144  break;
145  }
146 
147  if(!foundNextEdge) {
148  // Polygon incomplete, let's go to the next one
149  goto nextPolygon;
150  } else {
151  p.edges.push_back(nextEdge);
152 
153  if(nextEdge.b == firstPoint) {
154  // Polygon complete
155  polygons[it->first].push_back(p);
156  goto nextPolygon;
157  } else {
158  // Polygon in progress
159  searchPoint = nextEdge.b;
160  continue;
161  }
162  }
163 
164  nextPolygon:
165  p.edges.clear();
166  firstPoint = searchPoint = pEdges[0].a;
167  }
168  }
169  }
